# F14: Performance and Scaling Patterns for Financial Platforms

## Executive Summary

This document covers performance optimization strategies for a modular cashflow platform handling millions of journal entries and thousands of concurrent users. Research focuses on PostgreSQL indexing, caching, connection pooling, partitioning, and background job processing.

---

## 1. Database Indexing Strategies for Financial Queries

### Composite Indexes for Balance Sheet Queries

```sql
-- Optimal composite index for balance sheet queries
CREATE INDEX idx_journal_entries_account_period 
ON journal_entries (account_id, posting_date DESC, entry_id);

-- For trial balance with account hierarchy
CREATE INDEX idx_journal_entries_account_type_period 
ON journal_entries (account_type, account_id, posting_date)
INCLUDE (debit_amount, credit_amount);

-- Covering index for P&L queries (index-only scan)
CREATE INDEX idx_journal_entries_income_expense 
ON journal_entries (account_type, posting_date)
INCLUDE (debit_amount, credit_amount, department_id)
WHERE account_type IN ('REVENUE', 'EXPENSE');
```

### Partial Indexes for Module Filtering

```sql
-- Partial index for unreconciled entries (small subset, frequently queried)
CREATE INDEX idx_unreconciled_entries 
ON journal_entries (account_id, posting_date)
WHERE reconciliation_status = 'UNRECONCILED';

-- Partial index for draft entries (active workflow)
CREATE INDEX idx_draft_entries 
ON journal_entries (created_by, created_at)
WHERE status = 'DRAFT';

-- Partial index for active accounts only
CREATE INDEX idx_active_accounts 
ON accounts (account_code, account_name)
WHERE is_active = true AND is_archived = false;
```

### GIN Indexes for JSONB Metadata

```sql
-- GIN index for flexible metadata queries
CREATE INDEX idx_journal_entries_metadata 
ON journal_entries USING GIN (metadata jsonb_path_ops);

-- For querying specific metadata fields
CREATE INDEX idx_journal_entries_cost_center 
ON journal_entries USING GIN ((metadata->'cost_center'));

-- Composite GIN for complex metadata queries
CREATE INDEX idx_journal_entries_metadata_composite 
ON journal_entries USING GIN (metadata, tags);
```

**Key Insight**: Partial indexes reduce index size by 60-80% when filtering on low-selectivity columns (e.g., status='DRAFT' when only 5% of entries are drafts). GIN indexes on JSONB add ~10-15% write overhead but enable flexible querying without schema changes.

## 2. Query Optimization for Financial Reports

### Materialized Views vs Live Aggregation

**Materialized Views (Recommended for dashboards)**:

```sql
-- Daily balance summary (refresh nightly)
CREATE MATERIALIZED VIEW mv_daily_balances AS
SELECT 
    account_id,
    DATE(posting_date) as balance_date,
    SUM(debit_amount) as total_debits,
    SUM(credit_amount) as total_credits,
    SUM(debit_amount - credit_amount) as net_amount,
    COUNT(*) as entry_count
FROM journal_entries
WHERE status = 'POSTED'
GROUP BY account_id, DATE(posting_date);

CREATE UNIQUE INDEX idx_mv_daily_balances 
ON mv_daily_balances (account_id, balance_date);

-- Monthly trial balance (refresh hourly during business hours)
CREATE MATERIALIZED VIEW mv_monthly_trial_balance AS
SELECT 
    account_id,
    account_code,
    account_name,
    account_type,
    DATE_TRUNC('month', posting_date) as period,
    SUM(CASE WHEN debit_amount > 0 THEN debit_amount ELSE 0 END) as total_debits,
    SUM(CASE WHEN credit_amount > 0 THEN credit_amount ELSE 0 END) as total_credits,
    SUM(debit_amount - credit_amount) as balance
FROM journal_entries
JOIN accounts ON journal_entries.account_id = accounts.id
WHERE status = 'POSTED'
GROUP BY account_id, account_code, account_name, account_type, DATE_TRUNC('month', posting_date);
```

**Live Aggregation (for real-time queries)**:

```sql
-- Optimized live query for current balance
SELECT 
    a.id,
    a.account_code,
    a.account_name,
    COALESCE(SUM(e.debit_amount - e.credit_amount), 0) as current_balance
FROM accounts a
LEFT JOIN journal_entries e ON a.id = e.account_id 
    AND e.status = 'POSTED'
    AND e.posting_date <= CURRENT_DATE
WHERE a.is_active = true
GROUP BY a.id, a.account_code, a.account_name;
```

### Refresh Strategies

```sql
-- Concurrent refresh (allows reads during refresh)
REFRESH MATERIALIZED VIEW CONCURRENTLY mv_daily_balances;

-- Scheduled refresh via pg_cron
SELECT cron.schedule('refresh-daily-balances', '0 2 * * *', 
    'REFRESH MATERIALIZED VIEW CONCURRENTLY mv_daily_balances');

-- Partial refresh for incremental updates
-- (Requires manual implementation with change tracking)
```

**Benchmark**: Materialized views reduce query time from 2-5 seconds to 50-200ms for complex financial reports. Trade-off: data freshness (staleness window of refresh interval).

### Account Balance Caching Strategy

```typescript
// Cache-aside pattern for balances
async function getAccountBalance(accountId: string): Promise<Balance> {
  const cacheKey = `balance:${accountId}`;
  
  // Try cache first
  const cached = await redis.get(cacheKey);
  if (cached) {
    return JSON.parse(cached);
  }
  
  // Cache miss - query database
  const balance = await db.query(`
    SELECT SUM(debit_amount - credit_amount) as balance
    FROM journal_entries
    WHERE account_id = $1 AND status = 'POSTED'
  `, [accountId]);
  
  // Store in cache with TTL
  await redis.setex(cacheKey, 300, JSON.stringify(balance));
  
  return balance;
}

// Invalidate on new entry
async function invalidateBalanceCache(accountId: string): Promise<void> {
  await redis.del(`balance:${accountId}`);
  // Also invalidate parent account caches
  await redis.del(`balance:parent:${accountId}`);
}
```

### Cache Invalidation Strategies for Financial Data

```typescript
// Tag-based invalidation
const invalidationTags = {
```

### Lag Tolerance for Financial Dashboards

```typescript
// Read replica routing with lag awareness
async function queryWithReplicaFallback<T>(
  query: string,
  params: any[]
): Promise<T> {
  // Check replica lag
  const lag = await getReplicaLag();
  
  if (lag < 5) { // Less than 5 seconds lag
    // Use replica for read queries
    return await replicaDb.query(query, params);
  } else {
    // Fall back to primary
    return await primaryDb.query(query, params);
  }
}

// Monitor replication lag
async function getReplicaLag(): Promise<number> {
  const result = await primaryDb.query(`
    SELECT EXTRACT(EPOCH FROM (now() - pg_last_xact_replay_timestamp()))::int as lag_seconds
  `);
  return result.rows[0].lag_seconds || 0;
}
```

**Lag Tolerance Guidelines**:
- Real-time balances: 0-2 seconds (use primary)
- Dashboard widgets: 5-30 seconds (acceptable for replicas)
- Historical reports: 1-5 minutes (replica preferred)
- Audit logs: 10-30 minutes (replica preferred)

## 6. Partitioning Strategies

### Time-Based Partitioning for Journal Entries

```sql
-- Monthly partitioning for journal_entries
CREATE TABLE journal_entries (
    id UUID DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    entry_date DATE NOT NULL,
    posting_date DATE NOT NULL,
    account_id UUID NOT NULL,
    debit_amount NUMERIC(18,2),
    credit_amount NUMERIC(18,2),
    status VARCHAR(20) NOT NULL DEFAULT 'DRAFT',
    created_at TIMESTAMP DEFAULT NOW(),
    PRIMARY KEY (id, posting_date)
) PARTITION BY RANGE (posting_date);

-- Create monthly partitions
CREATE TABLE journal_entries_2024_01 PARTITION OF journal_entries
    FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

CREATE TABLE journal_entries_2024_02 PARTITION OF journal_entries
    FOR VALUES FROM ('2024-02-01') TO ('2024-03-01');

-- Automate partition creation
CREATE OR REPLACE FUNCTION create_monthly_partition()
RETURNS void AS $$
DECLARE
    next_month DATE := DATE_TRUNC('month', NOW() + INTERVAL '1 month');
    partition_name TEXT;
BEGIN
    partition_name := 'journal_entries_' || 
        TO_CHAR(next_month, 'YYYY_MM');
    
    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF journal_entries
         FOR VALUES FROM (%L) TO (%L)',
        partition_name,
        next_month,
        next_month + INTERVAL '1 month'
    );
END;
$$ LANGUAGE plpgsql;
```

### Tenant-Based Partitioning for Multi-Tenant

```sql
-- List partitioning by tenant
CREATE TABLE journal_entries (
    id UUID DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    entry_date DATE NOT NULL,
    account_id UUID NOT NULL,
    debit_amount NUMERIC(18,2),
    credit_amount NUMERIC(18,2),
    PRIMARY KEY (id, tenant_id)
) PARTITION BY LIST (tenant_id);

-- Create tenant partitions
CREATE TABLE journal_entries_tenant_a PARTITION OF journal_entries
    FOR VALUES IN ('tenant-a-uuid');

CREATE TABLE journal_entries_tenant_b PARTITION OF journal_entries
    FOR VALUES IN ('tenant-b-uuid');
```

### Hybrid Partitioning Strategy

```sql
-- Composite partitioning: tenant + time
CREATE TABLE journal_entries (
    id UUID DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL,
    posting_date DATE NOT NULL,
    account_id UUID NOT NULL,
    debit_amount NUMERIC(18,2),
    credit_amount NUMERIC(18,2),
    PRIMARY KEY (id, tenant_id, posting_date)
) PARTITION BY LIST (tenant_id);

-- Sub-partition each tenant by time
CREATE TABLE journal_entries_tenant_a PARTITION OF journal_entries
    FOR VALUES IN ('tenant-a-uuid')
    PARTITION BY RANGE (posting_date);
```

**Partition Pruning**: PostgreSQL automatically excludes partitions not matching query predicates. Enable with `enable_partition_pruning = on` (default).

**Maintenance**: Drop old partitions with `DROP TABLE` (instant) instead of `DELETE` (slow, causes vacuum).

## 7. Bulk Operations

### Batch Journal Entry Posting

```typescript
// Batch insert with optimal performance
async function batchPostEntries(entries: JournalEntry[]): Promise<void> {
  const batchSize = 1000;
  
  for (let i = 0; i < entries.length; i += batchSize) {
    const batch = entries.slice(i, i + batchSize);
    
    await db.transaction(async (trx) => {
      // Use COPY for maximum performance
      await trx.raw(`
        COPY journal_entries (tenant_id, entry_date, posting_date, 
          account_id, debit_amount, credit_amount, status, created_by)
        FROM STDIN
      `);
      
      // Or use multi-row INSERT (alternative)
      const values = batch.map(e => 
        `('${e.tenantId}', '${e.entryDate}', '${e.postingDate}', 
         '${e.accountId}', ${e.debitAmount}, ${e.creditAmount}, 
         'POSTED', '${e.createdBy}')`
      ).join(',');
      
      await trx.raw(`
        INSERT INTO journal_entries 
        (tenant_id, entry_date, posting_date, account_id, 
         debit_amount, credit_amount, status, created_by)
        VALUES ${values}
      `);
    });
  }
}
```

### Bulk Import Performance

```typescript
// Optimal bulk import strategy
async function bulkImportJournalEntries(file: Buffer): Promise<void> {
  // 1. Load into staging table (minimal indexes)
  await db.raw(`
    COPY journal_entries_staging (tenant_id, entry_date, posting_date,
      account_id, debit_amount, credit_amount)
    FROM STDIN
  `);
  
  // 2. Validate and transform in batch
  await db.raw(`
    INSERT INTO journal_entries_staging_validated
    SELECT * FROM journal_entries_staging
    WHERE validate_entry(account_id, debit_amount, credit_amount)
  `);
  
  // 3. Insert into main table with minimal locking
  await db.raw(`
    INSERT INTO journal_entries 
    SELECT gen_random_uuid(), tenant_id, entry_date, posting_date,
           account_id, debit_amount, credit_amount, 'POSTED', NOW()
    FROM journal_entries_staging_validated
    ON CONFLICT DO NOTHING
  `);
  
  // 4. Update materialized views
  await db.raw('REFRESH MATERIALIZED VIEW CONCURRENTLY mv_daily_balances');
}
```

**Bulk Operation Benchmarks**:
- Single INSERT: ~500 rows/sec
- Multi-row INSERT (100 rows): ~5,000 rows/sec
- COPY command: ~50,000-100,000 rows/sec
- Batch size optimal: 1,000-10,000 rows

## 9. Background Job Processing with BullMQ

### Queue Configuration for Financial Systems

```typescript
import { Queue, Worker } from 'bullmq';

// Journal entry posting queue
const journalQueue = new Queue('journal-entries', {
  connection: {
    host: 'redis-host',
    port: 6379,
  },
  defaultJobOptions: {
    attempts: 3,
    backoff: {
      type: 'exponential',
      delay: 1000,
    },
    removeOnComplete: {
      age: 86400,    // Keep completed jobs for 24 hours
      count: 1000,   // Keep last 1000 completed jobs
    },
    removeOnFail: {
      age: 604800,   // Keep failed jobs for 7 days
    },
  },
});

// Worker for processing journal entries
const journalWorker = new Worker('journal-entries', async (job) => {
  const { entries, tenantId, userId } = job.data;
  
  // Update job progress
  await job.updateProgress(0);
  
  // Process entries in batches
  const batchSize = 100;
  for (let i = 0; i < entries.length; i += batchSize) {
    const batch = entries.slice(i, i + batchSize);
    
    await processBatch(batch, tenantId, userId);
    
    // Update progress
    await job.updateProgress(Math.floor((i / entries.length) * 100));
  }
  
  return { processed: entries.length };
}, {
  connection: {
    host: 'redis-host',
    port: 6379,
  },
  concurrency: 5,  // Process 5 jobs simultaneously
  limiter: {
    max: 100,       // Max 100 jobs
    duration: 1000, // per second
  },
});

// Dead letter queue for failed jobs
const deadLetterQueue = new Queue('journal-entries-dlq', {
  connection: {
    host: 'redis-host',
    port: 6379,
  },
});

journalWorker.on('failed', async (job, error) => {
  if (job.attemptsMade >= job.opts.attempts) {
    // Move to dead letter queue
    await deadLetterQueue.add('failed-entry', {
      jobId: job.id,
      data: job.data,
      error: error.message,
      failedAt: new Date(),
    });
  }
});
```
